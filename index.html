google.charts.load('current', {'packages':['sankey']}); google.charts.setOnLoadCallback(drawChart); function drawChart() { var data = new google.visualization.DataTable(); data.addColumn('string', 'From'); data.addColumn('string', 'To'); data.addColumn('number', 'Weight'); data.addRows([ [ 'Revenue', 'Fees', 500 ], [ 'Fees', 'Marketing', 50 ], [ 'Fees', 'House Wallet', 500*.25/3.5 ], [ 'Fees', 'Dev Team', 450*.25/3.5 ], [ 'Fees', 'Coin Holders', 450-500*.25/3.5-450*.25/3.5 ] ]); // Sets chart options. var options = { width: 600, }; // Instantiates and draws our chart, passing in some options. var chart = new google.visualization.Sankey(document.getElementById('sankey_basic')); chart.draw(data, options); }
